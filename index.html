#!/usr/bin/env python
from __future__ import print_function
import sys
import csv

reader = csv.reader(sys.stdin)
for row in reader:
    if not row:
        continue
    # expected columns: State, PollingStation, Candidate, Votes
    if len(row) < 4:
        continue
    state = row[0].strip()
    candidate = row[2].strip()
    votes = row[3].strip()
    # skip header
    if state.lower() == 'state' and candidate.lower() == 'candidate':
        continue
    try:
        v = int(votes)
    except:
        continue
    # Emit key as "State|Candidate" and value as votes
    key = state + '|' + candidate
    print("{0}\t{1}".format(key, v))
